; 11. Write a function to determine the depth of a list.
(DEFUN getDepthOfList (x currentLevel)
       (COND
        ((atom x) currentLevel)
        (t (apply #'max (mapcar #' (lambda (y) (getDepthOfList y (+ currentLevel 1))) x)))
         )
)

(print (getDepthOfList '(1 (2 (5 (9 (12 (0 2 1 5 (3) (-2 1) (7 1 (-3)))))))) 0))
