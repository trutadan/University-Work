; 8. Write a function to determine the number of nodes on the level k from a n-tree represented as follows: (rootlist_nodes_subtree1 ... list_nodes_subtreen).
; Eg: tree is (a (b (c)) (d) (e (f))) and k=1 => 3 nodes.
(DEFUN getNumberOfNodesOnGivenLevel (l currentLevel givenLevel)
           (COND
            ((null l) 0)
            ((and (atom l) (equal currentLevel givenLevel)) 1)
            ((atom l) 0)
            (t (apply #'+ (mapcar #' (lambda (l)  (getNumberOfNodesOnGivenLevel l (+ currentLevel 1) givenLevel)) l)))
           )       
)

(DEFUN mainGetNumberOfNodesOnGivenLevel (l givenLevel)
     (getNumberOfNodesOnGivenLevel l 0 givenLevel)
)

(print(mainGetNumberOfNodesOnGivenLevel '(a (b (c)) (d) (e (f))) 2))
