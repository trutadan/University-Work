; 14. Write a function that returns the number of atoms in a list, at any level. 
(DEFUN getNumberOfAtoms (x) 
        (COND
         ((null x) 0)
         ((atom x) 1)
         (t (apply #'+ (mapcar #' getNumberOfAtoms x)))
         )
)

(print(getNumberOfAtoms '(-1 -2 -3 -4 (-1 ((-2) -5) -3))))
(print(getNumberOfAtoms '()))
