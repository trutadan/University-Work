; 10. Define a function that replaces one node with another one in a n-tree represented as: root 
; list_of_nodes_subtree1... list_of_nodes_subtreen) 
; Eg: tree is (a (b (c)) (d) (e (f))) and node 'b' will be replace with node 'g' => tree (a (g (c)) (d) (e (f)))
(DEFUN replaceNodeWithAnother (tree currentNode replacementNode)
       (COND
        ((and (atom tree) (equal tree currentNode)) replacementNode)
        ((atom tree) tree)
        (t (mapcar #' (lambda (x) (replaceNodeWithAnother x currentNode replacementNode)) tree))
        )
)

(print (replaceNodeWithAnother '(a (b (c)) (d) (e (f))) 'b 'g))
