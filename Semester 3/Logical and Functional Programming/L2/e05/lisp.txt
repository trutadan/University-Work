(DEFUN differenceBetweenSumOfEvensAndSumOfOdds (l)
         (COND
          ((null l) 0)
          ((and (numberp l) (equal (mod l 2) 1)) (* l -1))
          ((and (numberp l) (equal (mod l 2) 0)) l)
          (t (apply #'+ (mapcar #'differenceBetweenSumOfEvensAndSumOfOdds l)))
          )
)
