; 7. Write a function that substitutes an element E with all elements of a list L1 at all levels of a given list L.
(DEFUN substituteElementWithList (l element sl)
       (COND
        ((null l) nil)
        ((and (atom l) (= l element)) sl)
        ((atom l) l)
        (t (car (list (mapcar #' (lambda (l) (substituteElementWithList l element sl)) l))))
        )
)

(print (substituteElementWithList '(1 2 3 (1 2 (1))) '1 '(1 2 3)))
