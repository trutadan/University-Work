start
puts("Give me a natural number");
int a;
a = to_i(cuts(gets()));

puts("Give me another natural number");
int b;
b = to_i(cuts(gets()));

if a < b
    int aux;
    aux = a;
    a = b;
    b = aux;
end

while b != 0 do
    int aux;
    aux = b;
    b = a % b;
    a = aux;
end

puts(`The GCD of the given numbers is #{a}`);
stop
